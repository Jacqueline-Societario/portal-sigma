POST  https://societario.gsigma.com.br/webhook/forms/<SEU_TOKEN_AQUI>
Script Google Apps Script

Para cada formulário Google, siga os passos:
1. Abra o formulário no Google Forms
2. Clique nos 3 pontos no topo direito → Script editor
3. Cole o script abaixo, ajuste o token e salve
4. Vá em Triggers (ícone de relógio) → Adicione gatilho: função onFormSubmit, evento On form submit

// ═══════════════════════════════════════════════════════════════ // Portal Societário Sigma — Integração Google Forms // Cole este script no editor do Google Apps Script do formulário. // ═══════════════════════════════════════════════════════════════ var PORTAL_URL = "https://societario.gsigma.com.br/webhook/forms/SEU_TOKEN_AQUI"; var FORM_NAME = "Entrevista de Abertura de Empresa"; // Altere para o nome deste formulário var TIPO_PROCESSO = "Abertura de Empresa"; // Tipo de processo (ex: Alteração, MEI, Alvará) function onFormSubmit(e) { try { var form = FormApp.getActiveForm(); var response = e.response; // Montar payload var payload = { form_id: form.getId(), form_name: FORM_NAME, tipo_processo: TIPO_PROCESSO, response_id: response.getId(), respondent_email: response.getRespondentEmail() || "", submitted_at: response.getTimestamp().toISOString(), responses: {} }; // Coletar respostas (pergunta → resposta) var itemResponses = response.getItemResponses(); for (var i = 0; i < itemResponses.length; i++) { var item = itemResponses[i]; var pergunta = item.getItem().getTitle(); var resposta = item.getResponse(); // Grids e caixas de seleção múltipla retornam array if (Array.isArray(resposta)) { resposta = resposta.join(", "); } payload.responses[pergunta] = resposta || ""; } // Enviar para o portal var options = { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true }; var result = UrlFetchApp.fetch(PORTAL_URL, options); Logger.log("Portal response: " + result.getResponseCode() + " — " + result.getContentText()); } catch (err) { Logger.log("Erro ao enviar para o portal: " + err.toString()); } }
